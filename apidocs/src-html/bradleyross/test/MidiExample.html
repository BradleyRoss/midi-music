001package bradleyross.test;
002import java.util.List;
003import javax.sound.midi.MidiDeviceReceiver;
004import javax.sound.midi.MidiDeviceTransmitter;
005import javax.sound.midi.MidiMessage;
006import javax.sound.midi.MidiEvent;
007import javax.sound.midi.Receiver;
008import javax.sound.midi.Sequencer;
009import javax.sound.midi.Sequence;
010import javax.sound.midi.Synthesizer;
011import javax.sound.midi.Track;
012import javax.sound.midi.Transmitter;
013import javax.sound.midi.MidiSystem;
014import javax.sound.midi.MidiDevice;
015import javax.sound.midi.MidiDevice.Info;
016import javax.sound.midi.MidiUnavailableException;
017import javax.sound.midi.InvalidMidiDataException;
018/**
019 * Demonstration of showing key clicks from a MIDI device.
020 * 
021 * <p>The Javadocs for the MIDI java classes can be found at 
022 * <a href="https://docs.oracle.com/javase/7/docs/api/overview-summary.html" target="_blank">
023 * https://docs.oracle.com/javase/7/docs/api/overview-summary.html</a><p>
024 * <ul>
025 * <li><a href="https://www.ibm.com/developerworks/library/it/it-0801art38/" target="_blank">
026 *     https://www.ibm.com/developerworks/library/it/it-0801art38/</a> IBM article on 
027 *     writing programs for MIDI</li>
028 * </ul>
029 * @author Bradley Ross
030 *
031 */
032public class MidiExample implements Runnable {
033        protected Synthesizer synthesizer = null;
034        protected Sequencer sequencer = null;
035        public void run() {
036                MidiDevice.Info[] list = MidiSystem.getMidiDeviceInfo();
037                Synthesizer synthesizer = null;
038
039                MidiDevice keyboard = null;
040                MidiDevice device = null;
041                try {
042                        synthesizer = MidiSystem.getSynthesizer();
043                        sequencer = MidiSystem.getSequencer(true);
044                } catch (MidiUnavailableException e) {
045                        System.out.println("Failure setting up sequencer/synthesizer");
046                        e.printStackTrace();
047                }
048                System.out.println("I, Vendor, Name, Description, Version");
049                for (int i = 0; i < list.length; i++)  {
050                        System.out.println("*****  *****");
051                        Info item = list[i];
052
053                        try {
054
055
056                                device = MidiSystem.getMidiDevice(item);
057                                System.out.println(i + " : " + item.getVendor() + " : " + item.getName() + " : " +
058                                                list[i].getDescription() + " : " + list[i].getVersion());
059                                System.out.println(device.getClass().getName());
060                                // device.open();
061                                if (device instanceof Sequencer) {
062                                        System.out.println("subclass of Sequencer"); 
063                                        // sequencer = (Sequencer)device;
064                                        int rcv = device.getMaxReceivers();
065                                        int tx = device.getMaxTransmitters();
066                                        System.out.println("Maximum of " + rcv + " receivers");
067                                        System.out.println("Maximum of " + tx + " transmitters");                                
068                                } else if (device instanceof Synthesizer) {
069                                        System.out.println("subclass of Synthesizer"); 
070                                        synthesizer = (Synthesizer)device;
071                                        int rcv = device.getMaxReceivers();
072                                        int tx = device.getMaxTransmitters();
073                                        System.out.println("Maximum of " + rcv + " receivers");
074                                        System.out.println("Maximum of " + tx + " transmitters");
075                                } else {
076                                        System.out.println("Device is a MIDI port");
077                                        int rcv = device.getMaxReceivers();
078                                        int tx = device.getMaxTransmitters();
079                                        System.out.println("Maximum of " + rcv + " receivers");
080                                        System.out.println("Maximum of " + tx + " transmitters");       
081
082                                        if (device instanceof MidiDeviceReceiver) {
083                                                System.out.println("subclass of MidiDeviceReceiver");
084                                        } 
085                                        if (device instanceof MidiDeviceTransmitter) {
086                                                System.out.println("subclass of MidiDeviceTransmitter");
087                                        }
088                                        if (device instanceof Receiver) {
089                                                System.out.println("subclass of Receiver");
090                                        } 
091                                        if (device instanceof Transmitter) {
092                                                System.out.println("subclass of Transmitter"); 
093                                        } 
094                                        if (device instanceof MidiDevice) {
095                                                System.out.println("subclass of MidiDevice");
096                                        } else {
097                                                System.out.println("Unknown type");
098                                        }
099                                        if (item.getName().contains("KEYBOARD")) {
100                                                keyboard = device;
101                                        }
102                                        List<Receiver>rcvList = device.getReceivers();
103                                        List<Transmitter>txList = device.getTransmitters();
104                                        if (rcvList != null) {
105                                                System.out.println("There are " + rcvList.size() + " receivers");
106                                        }
107                                        if (txList != null) {
108                                                System.out.println("There are " + txList.size() + " transmitters");
109                                        }
110                                        device.close();
111                                }
112
113                        }
114                        catch (MidiUnavailableException e) {
115                                System.out.println("MidiUnavailableException");
116                                e.printStackTrace();
117                        }
118                }
119                Transmitter inPortTrans1, inPortTrans2;
120                Receiver synthRcvr;
121                Receiver seqRcvr;
122                
123                if (keyboard != null) {
124                        try {
125                                System.out.println(keyboard.getDeviceInfo().getName());
126                                inPortTrans1 = keyboard.getTransmitter();
127                                synthRcvr = synthesizer.getReceiver();
128                                inPortTrans1.setReceiver(synthRcvr);
129                                inPortTrans2 = keyboard.getTransmitter();
130                                seqRcvr = sequencer.getReceiver();
131                                inPortTrans2.setReceiver(seqRcvr);
132                                for (int i = 0; i < 5; i++) {
133                                        System.out.println("Sleeping");
134                                
135                                Thread.sleep(50000);
136                        }
137                                System.out.println("Sleep complete");
138                        } catch (MidiUnavailableException e) {
139                                e.printStackTrace();
140                        } catch (InterruptedException e) {
141                                e.printStackTrace();
142                        } /* catch (InvalidMidiDataException e) {
143                                e.printStackTrace();
144                        } */
145                        
146                }
147                
148        }
149        public static void main(String[] args) {
150                
151                MidiExample instance = new MidiExample();
152                instance.run();
153        }
154}
