047/**
048 * This class is a Swing component that can load and play a sound clip,
049 * displaying progress and controls.  The main( ) method is a test program.
050 * This component can play sampled audio or MIDI files, but handles them 
051 * differently. For sampled audio, time is reported in microseconds, tracked in
052 * milliseconds and displayed in seconds and tenths of seconds. For midi
053 * files time is reported, tracked, and displayed in MIDI "ticks".
054 * This program does no transcoding, so it can only play sound files that use
055 * the PCM encoding.
056 * 
057 * <p>The first and only parameter for the main method is the name of the file
058 *    that provides the audio or midi sample.</p>
059 * <p>This is taken from Section 17.4 of the Third Edition of Java Examples in a Nutshell,
060 *    "Playing sounds with javax.sound".</p>
061 */
062@SuppressWarnings("serial")
063public class SoundPlayer extends JComponent {
064    boolean midi;            // Are we playing a midi file or a sampled one?
065    Sequence sequence;       // The contents of a MIDI file
066    Sequencer sequencer;     // We play MIDI Sequences with a Sequencer
067    Clip clip;               // Contents of a sampled audio file
068    boolean playing = false; // whether the sound is currently playing
069
070    // Length and position of the sound are measured in milliseconds for 
071    // sampled sounds and MIDI "ticks" for MIDI sounds
072    int audioLength;         // Length of the sound.  
073    int audioPosition = 0;   // Current position within the sound
074
075    // The following fields are for the GUI
076    JButton play;             // The Play/Stop button
077    JSlider progress;         // Shows and sets current position in sound
078    JLabel time;              // Displays audioPosition as a number
079    Timer timer;              // Updates slider every 100 milliseconds
080
081    // The main method just creates a SoundPlayer in a Frame and displays it
082    public static void main(String[  ] args) 
083        throws IOException,
084               UnsupportedAudioFileException,
085               LineUnavailableException,
086               MidiUnavailableException,
087               InvalidMidiDataException
088    {
089        SoundPlayer player;
090        File file;
091        if (args.length == 0) {
092                file=new File("/Users/bradleyross/Downloads/AGNICRT.MID");
093        } else {
094        file = new File(args[0]); 
095        } 
096        // This is the file we'll be playing
097        // Determine whether it is midi or sampled audio
098        boolean ismidi;
099        try {
100            // We discard the return value of this method; we just need to know
101            // whether it returns successfully or throws an exception
102            MidiSystem.getMidiFileFormat(file);
103            ismidi = true;
104        }
105        catch(InvalidMidiDataException e) {
106            ismidi = false;
107        }
108
109        // Create a SoundPlayer object to play the sound.
110        player = new SoundPlayer(file, ismidi);
111
112        // Put it in a window and play it
113        JFrame f = new JFrame("SoundPlayer");
114        f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
115        f.getContentPane( ).add(player, "Center");
116        f.pack( );
117        f.setVisible(true);
118    }

120    // Create a SoundPlayer component for the specified file.
121    public SoundPlayer(File f, boolean isMidi)
122        throws IOException,
123               UnsupportedAudioFileException,
124               LineUnavailableException,
125               MidiUnavailableException,
126               InvalidMidiDataException
127    {
128        if (isMidi) {     // The file is a MIDI file
129            midi = true;
130            // First, get a Sequencer to play sequences of MIDI events
131            // That is, to send events to a Synthesizer at the right time.
132            sequencer = MidiSystem.getSequencer( );  // Used to play sequences
133            sequencer.open( );                       // Turn it on.
134
135            // Get a Synthesizer for the Sequencer to send notes to
136            Synthesizer synth = MidiSystem.getSynthesizer( );
137            synth.open( );  // acquire whatever resources it needs
138            
139            // The Sequencer obtained above may be connected to a Synthesizer
140            // by default, or it may not.  Therefore, we explicitly connect it.
141            Transmitter transmitter = sequencer.getTransmitter( );
142            Receiver receiver = synth.getReceiver( );
143            transmitter.setReceiver(receiver);
144            
145            // Read the sequence from the file and tell the sequencer about it
146            sequence = MidiSystem.getSequence(f);
147            sequencer.setSequence(sequence);
148            audioLength = (int)sequence.getTickLength( ); // Get sequence length
149        }
150        else {            // The file is sampled audio
151            midi = false;
152            // Getting a Clip object for a file of sampled audio data is kind
153            // of cumbersome.  The following lines do what we need.
154            AudioInputStream ain = AudioSystem.getAudioInputStream(f);
155            try {
156                DataLine.Info info =
157                    new DataLine.Info(Clip.class,ain.getFormat( ));
158                clip = (Clip) AudioSystem.getLine(info);
159                clip.open(ain);
160            }
161            finally { // We're done with the input stream.
162                ain.close( );
163            }
164            // Get the clip length in microseconds and convert to milliseconds
165            audioLength = (int)(clip.getMicrosecondLength( )/1000);
166        }
167
168        // Now create the basic GUI
169        play = new JButton("Play");                // Play/stop button
170        progress = new JSlider(0, audioLength, 0); // Shows position in sound
171        time = new JLabel("0");                    // Shows position as a #
172
173        // When clicked, start or stop playing the sound
174        play.addActionListener(new ActionListener( ) {
175                public void actionPerformed(ActionEvent e) {
176                    if (playing) stop( ); else play( );
177                }
178            });
179
180        // Whenever the slider value changes, first update the time label.
181        // Next, if we're not already at the new position, skip to it.
182        progress.addChangeListener(new ChangeListener( ) {
183                public void stateChanged(ChangeEvent e) {
184                    int value = progress.getValue( );
185                    // Update the time label
186                    if (midi) time.setText(value + "");
187                    else time.setText(value/1000 + "." +
188                                      (value%1000)/100);
189                    // If we're not already there, skip there.
190                    if (value != audioPosition) skip(value);
191                }
192            });
193        
194        // This timer calls the tick( ) method 10 times a second to keep 
195        // our slider in sync with the music.
196        timer = new javax.swing.Timer(100, new ActionListener( ) {
197                public void actionPerformed(ActionEvent e) { tick( ); }
198            });
199        
200        // put those controls in a row
201        Box row = Box.createHorizontalBox( );
202        row.add(play);
203        row.add(progress);
204        row.add(time);
205        
206        // And add them to this component.
207        setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
208        this.add(row);
209
210        // Now add additional controls based on the type of the sound
211        if (midi) addMidiControls( );
212        else addSampledControls( );
213    }
214
215    /** Start playing the sound at the current position */
216    public void play( ) {
217        if (midi) sequencer.start( );
218        else clip.start( );
219        timer.start( );
220        play.setText("Stop");
221        playing = true;
222    }
223
224    /** Stop playing the sound, but retain the current position */
225    public void stop( ) {
226        timer.stop( );
227        if (midi) sequencer.stop( );
228        else clip.stop( );
229        play.setText("Play");
230        playing = false;
231    }
232
233    /** Stop playing the sound and reset the position to 0 */
234    public void reset( ) {
235        stop( );
236        if (midi) sequencer.setTickPosition(0);
237        else clip.setMicrosecondPosition(0);
238        audioPosition = 0; 
239        progress.setValue(0);
240    }
241
242    /** Skip to the specified position */
243    public void skip(int position) { // Called when user drags the slider
244        if (position < 0 || position > audioLength) return;
245        audioPosition = position;
246        if (midi) sequencer.setTickPosition(position);
247        else clip.setMicrosecondPosition(position * 1000);
248        progress.setValue(position); // in case skip( ) is called from outside
249    }
250
251    /** Return the length of the sound in ms or ticks */
252    public int getLength( ) { return audioLength; }
253
254    // An internal method that updates the progress bar.
255    // The Timer object calls it 10 times a second.
256    // If the sound has finished, it resets to the beginning
257    void tick( ) {
258        if (midi && sequencer.isRunning( )) {
259            audioPosition = (int)sequencer.getTickPosition( );
260            progress.setValue(audioPosition);
261        }
262        else if (!midi && clip.isActive( )) {
263            audioPosition = (int)(clip.getMicrosecondPosition( )/1000);
264            progress.setValue(audioPosition);
265        }
266        else reset( );  
267    }

269    // For sampled sounds, add sliders to control volume and balance
270    void addSampledControls( ) {
271        try {
272            FloatControl gainControl =
273                (FloatControl)clip.getControl(FloatControl.Type.MASTER_GAIN);
274            if (gainControl != null) this.add(createSlider(gainControl));
275        }
276        catch(IllegalArgumentException e) {
277            // If MASTER_GAIN volume control is unsupported, just skip it
278        }
279
280        try {
281            // FloatControl.Type.BALANCE is probably the correct control to
282            // use here, but it doesn't work for me, so I use PAN instead.
283            FloatControl panControl =
284                (FloatControl)clip.getControl(FloatControl.Type.PAN);
285            if (panControl != null) this.add(createSlider(panControl));
286        }
287        catch(IllegalArgumentException e) {  }
288    }

291    // Return a JSlider component to manipulate the supplied FloatControl
292    // for sampled audio.
293    JSlider createSlider(final FloatControl c) {
294        if (c == null) return null;
295        final JSlider s = new JSlider(0, 1000);
296        final float min = c.getMinimum( );
297        final float max = c.getMaximum( );
298        final float width = max-min;
299        float fval = c.getValue( );
300        s.setValue((int) ((fval-min)/width * 1000));
301
302        java.util.Hashtable labels = new java.util.Hashtable(3);
303        labels.put(new Integer(0), new JLabel(c.getMinLabel( )));
304        labels.put(new Integer(500), new JLabel(c.getMidLabel( )));
305        labels.put(new Integer(1000), new JLabel(c.getMaxLabel( )));
306        s.setLabelTable(labels);
307        s.setPaintLabels(true);
308
309        s.setBorder(new TitledBorder(c.getType( ).toString( ) + " " +
310                                     c.getUnits( )));
311
312        s.addChangeListener(new ChangeListener( ) {
313                public void stateChanged(ChangeEvent e) {
314                    int i = s.getValue( );
315                    float f = min + (i*width/1000.0f);
316                    c.setValue(f);
317                }
318            });
319        return s;
320    }

322    // For Midi files, create a JSlider to control the tempo,
323    // and create JCheckBoxes to mute or solo each MIDI track.
324    void addMidiControls( ) {
325        // Add a slider to control the tempo
326        final JSlider tempo = new JSlider(50, 200);
327        tempo.setValue((int)(sequencer.getTempoFactor( )*100));
328        tempo.setBorder(new TitledBorder("Tempo Adjustment (%)"));
329        java.util.Hashtable<Integer,JLabel> labels = new java.util.Hashtable<Integer,JLabel>( );
330        labels.put(new Integer(50), new JLabel("50%"));
331        labels.put(new Integer(100), new JLabel("100%"));
332        labels.put(new Integer(200), new JLabel("200%"));
333        tempo.setLabelTable(labels);
334        tempo.setPaintLabels(true);
335        // The event listener actually changes the tempo
336        tempo.addChangeListener(new ChangeListener( ) {
337                public void stateChanged(ChangeEvent e) {
338                    sequencer.setTempoFactor(tempo.getValue( )/100.0f);
339                }
340            });
341
342        this.add(tempo);
343
344        // Create rows of solo and checkboxes for each track
345        Track[  ] tracks = sequence.getTracks( );
346        for(int i = 0; i < tracks.length; i++) {
347            final int tracknum = i;
348            // Two checkboxes per track
349            final JCheckBox solo = new JCheckBox("solo");
350            final JCheckBox mute = new JCheckBox("mute");
351            // The listeners solo or mute the track
352            solo.addActionListener(new ActionListener( ) {
353                    public void actionPerformed(ActionEvent e) {
354                        sequencer.setTrackSolo(tracknum,solo.isSelected( ));
355                    }
356                });
357            mute.addActionListener(new ActionListener( ) {
358                    public void actionPerformed(ActionEvent e) {
359                        sequencer.setTrackMute(tracknum,mute.isSelected( ));
360                    }
361                });
362
363            // Build up a row
364            Box box = Box.createHorizontalBox( );
365            box.add(new JLabel("Track " + tracknum));
366            box.add(Box.createHorizontalStrut(10));
367            box.add(solo);
368            box.add(Box.createHorizontalStrut(10));
369            box.add(mute);
370            box.add(Box.createHorizontalGlue( ));
371            // And add it to this component
372            this.add(box);
373        }
374    }
