004package bradleyross.music.playpen;
005import javax.sound.midi.Instrument;
006import javax.sound.midi.InvalidMidiDataException;
007import javax.sound.midi.MidiChannel;
008import javax.sound.midi.MidiDevice;
009import javax.sound.midi.MidiDeviceReceiver;
010import javax.sound.midi.MidiSystem;
011import javax.sound.midi.MidiUnavailableException;
012import javax.sound.midi.Receiver;
013import javax.sound.midi.ShortMessage;
014import javax.sound.midi.Synthesizer;
015
016
017/**
018 * This contains some examples based on the 
019 * <a href="https://docs.oracle.com/javase/tutorial/sound/overview-MIDI.html" target="_blank">
020 * Oracle Midi tutorials</a>.
021 * 
022 * <p>The documents are at 
023 * @author Bradley Ross
024 *
025 */
026public class OracleExamples1 {
027        protected Synthesizer synth;
028        protected void sample1() {
029                System.out.println("Starting sample1");
030                try {
031                        MidiDeviceReceiver rcvr = (MidiDeviceReceiver)MidiSystem.getReceiver();
032                        MidiDevice dev = rcvr.getMidiDevice();
033                        dev.open();
034                        System.out.println("isOpen is " + dev.isOpen());
035                        System.out.println(rcvr.getMidiDevice().getDeviceInfo().getDescription());
036                        ShortMessage myMsg = new ShortMessage();
037                        // Start playing the note Middle C (60), 
038                        // moderately loud (velocity = 93).
039                        myMsg.setMessage(ShortMessage.NOTE_ON, 0, 60, 93);
040                        long timeStamp = -1;
041
042                        rcvr.send(myMsg, timeStamp);
043                } catch (InvalidMidiDataException e) {
044                        System.out.println("InvalidMidiDataException encountered");
045                        e.printStackTrace();
046                } catch (MidiUnavailableException e) {
047                        System.out.println("MidiUnavailableException encountered");
048                        e.printStackTrace();
049                }
050        }
051        protected void sample2() {
052                try {
053                        synth = MidiSystem.getSynthesizer();
054                        synth.getDefaultSoundbank();
055                        Instrument[] inst = synth.getLoadedInstruments();
056                        int counter = 0;
057                        for (Instrument i : inst) {
058                                counter++;
059                                System.out.println(Integer.toString(counter) + " " + i.getName());
060                        }
061                        counter = 0;
062                        inst = synth.getAvailableInstruments();
063                        for (Instrument i : inst) {
064                                counter++;
065                                System.out.println(Integer.toString(counter) + " " + i.getName());
066                                if (i.getName().equalsIgnoreCase("acoustic grand piano")) {
067                                        synth.loadInstrument(i);
068                                }
069                        }
070                        MidiChannel[] channels = synth.getChannels();
071                        for (MidiChannel item : channels) {
072                                System.out.println(item.getProgram());
073                        }
074                }catch (MidiUnavailableException e) {
075                        e.printStackTrace();
076                } catch (Exception e) {
077                        e.printStackTrace();
078                }
079        }
080        /**
081         * @param args
082         */
083        public static void main(String[] args) {
084                // TODO Auto-generated method stub
085                OracleExamples1 instance = new OracleExamples1();
086                instance.sample2();
087                instance.sample1();
088                System.out.println("Program terminated");
089        }
090
091}
