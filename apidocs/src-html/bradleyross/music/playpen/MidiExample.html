019/**
020 * I am trying to write code that will convert key presses on keyboard to music.
021 * 
022 * <p>The Javadocs for the MIDI java classes can be found at 
023 * <a href="https://docs.oracle.com/javase/7/docs/api/overview-summary.html" target="_blank">
024 * https://docs.oracle.com/javase/7/docs/api/overview-summary.html</a><p>
025 * <ul>
026 * <li><a href="https://www.ibm.com/developerworks/library/it/it-0801art38/" target="_blank">
027 *     https://www.ibm.com/developerworks/library/it/it-0801art38/</a> IBM article on 
028 *     writing programs for MIDI</li>
029 * </ul>
030 * @author Bradley Ross
031 *
032 */
033public class MidiExample implements Runnable {
034        protected Synthesizer synthesizer = null;
035        protected Sequencer sequencer = null;
036        public void run() {
037                MidiDevice.Info[] list = MidiSystem.getMidiDeviceInfo();
038                MidiDevice keyboard = null;
039                MidiDevice device = null;
040                try {
041                        synthesizer = MidiSystem.getSynthesizer();
042                        sequencer = MidiSystem.getSequencer(true);
043                } catch (MidiUnavailableException e) {
044                        System.out.println("Failure setting up sequencer/synthesizer");
045                        e.printStackTrace();
046                }
047                System.out.println("I, Vendor, Name, Description, Version");
048                for (int i = 0; i < list.length; i++)  {
049                        System.out.println("*****  *****");
050                        Info item = list[i];
051                        try {
052                                device = MidiSystem.getMidiDevice(item);
053                                System.out.println(i + " : " + item.getVendor() + " : " + item.getName() + " : " +
054                                                list[i].getDescription() + " : " + list[i].getVersion());
055                                System.out.println(device.getClass().getName());
056                                // device.open();
057                                if (device instanceof Sequencer) {
058                                        System.out.println("subclass of Sequencer"); 
059                                        // sequencer = (Sequencer)device;
060                                        int rcv = device.getMaxReceivers();
061                                        int tx = device.getMaxTransmitters();
062                                        System.out.println("Maximum of " + rcv + " receivers");
063                                        System.out.println("Maximum of " + tx + " transmitters");                                
064                                }  else if (device instanceof Synthesizer) {
065                                        System.out.println("subclass of Synthesizer"); 
066                                        synthesizer = (Synthesizer)device;
067                                        int rcv = device.getMaxReceivers();
068                                        int tx = device.getMaxTransmitters();
069                                        System.out.println("Maximum of " + rcv + " receivers");
070                                        System.out.println("Maximum of " + tx + " transmitters");
071                                } else {
072                                        System.out.println("Device is a MIDI port");
073                                        int rcv = device.getMaxReceivers();
074                                        int tx = device.getMaxTransmitters();
075                                        System.out.println("Maximum of " + rcv + " receivers");
076                                        System.out.println("Maximum of " + tx + " transmitters");
077                                }
078
079                                if (device instanceof MidiDeviceReceiver) {
080                                        System.out.println("subclass of MidiDeviceReceiver");
081                                } 
082                                if (device instanceof MidiDeviceTransmitter) {
083                                        System.out.println("subclass of MidiDeviceTransmitter");
084                                }
085                                if (device instanceof Receiver) {
086                                        System.out.println("subclass of Receiver");
087                                } 
088                                if (device instanceof Transmitter) {
089                                        System.out.println("subclass of Transmitter"); 
090                                } 
091                                if (device instanceof MidiDevice) {
092                                        System.out.println("subclass of MidiDevice");
093                                } 
094                                if (item.getName().contains("KEYBOARD")) {
095                                        keyboard = device;
096                                }
097                                List<Receiver> rcvList = device.getReceivers();
098                                List<Transmitter> txList = device.getTransmitters();
099                                if (rcvList != null) {
100                                        System.out.println("There are " + rcvList.size() + " receivers");
101                                }
102                                if (txList != null) {
103                                        System.out.println("There are " + txList.size() + " transmitters");
104                                }
105                        }
106                        catch (MidiUnavailableException e) {
107                                System.out.println("MidiUnavailableException");
108                                e.printStackTrace();
109                        }
110                }
111                if (keyboard == null) { 
112                        System.out.println("No keyboard");
113                        return;
114                }
115                Transmitter inPortTrans1, inPortTrans2;
116                Receiver synthRcvr;
117                Receiver seqRcvr;
118                try {
119                        System.out.println(keyboard.getDeviceInfo().getName());
120                        synthesizer.open();
121                        sequencer.open();
122                        inPortTrans1 = keyboard.getTransmitter();
123                        keyboard.open();
124                        synthRcvr = synthesizer.getReceiver();
125                        inPortTrans1.setReceiver(synthRcvr);
126                        inPortTrans2 = keyboard.getTransmitter();
127                        seqRcvr = sequencer.getReceiver();
128                        inPortTrans2.setReceiver(seqRcvr);
129                        transceiverData(inPortTrans1, "inPortTrans1");
130                        transceiverData(inPortTrans2, "inPortTrans2");
131                        transceiverData(synthRcvr, "synthRcvr");
132                        transceiverData(seqRcvr, "seqRcvr");
133
134                        /* Keyboard information */
135                        List<Receiver> rcv = keyboard.getReceivers();
136                        List<Transmitter> tx = keyboard.getTransmitters();
137                        System.out.println("Keyboard: " +
138                                        Integer.toString(rcv.size()) + " receivers, " +
139                                        Integer.toString(tx.size()) + " transmitters");
140                        /* Synthesizer information */
141                        rcv = synthesizer.getReceivers();
142                        tx = synthesizer.getTransmitters();
143                        System.out.println("Synthesizer: " + 
144                                        Integer.toString(rcv.size()) + " receivers, " +
145                                        Integer.toString(tx.size()) + " transmitters");                         
146                        rcv = sequencer.getReceivers();
147                        /* Sequencer information */
148                        tx = sequencer.getTransmitters();
149                        System.out.println("Sequencer: " + 
150                                        Integer.toString(rcv.size()) + " receivers, " +
151                                        Integer.toString(tx.size()) + " transmitters");                         
152                        /*
153                         * I am very confused about this part,  I want to ensure that the
154                         * application doesn't close before I can do some work. 
155                         */
156                        for (int i = 0; i < 5; i++) {
157                                System.out.println("Sleeping");
158                                Thread.sleep(50000);
159                        }
160                        System.out.println("Sleep complete");
161                        inPortTrans1.close();
162                        inPortTrans2.close();
163                        synthRcvr.close();
164                        seqRcvr.close();
165                        /*
166                        synthesizer.close();
167                        sequencer.close();
168                        keyboard.close();
169                        */
170                } catch (MidiUnavailableException e) {
171                        e.printStackTrace();
172                } catch (InterruptedException e) {
173                        e.printStackTrace();
174                } 
175        }
176        /**
177         * This is a convenience class that prints some information on 
178         * Transmitter and Receiver objects.
179         * @param obj object for which information is displayed
180         * @param name label for object
181         */
182        protected void transceiverData(Object obj, String name) {
183                System.out.println(name + " class is " + obj.getClass().getName());
184                if (obj instanceof MidiDeviceReceiver) {
185                        System.out.println("     Subclass of MidiDeviceReceiver");
186                }
187                if (obj instanceof MidiDeviceTransmitter) {
188                        System.out.println("     Subclass of MidiDeviceTransmitter");
189                }
190                if (obj instanceof Receiver) {
191                        System.out.println("     Subclass of Receiver");
192                }
193                if (obj instanceof Transmitter) {
194                        System.out.println("     Subclass of Transmitter");
195                }
196        }
197        /**
198         * Test driver.
199         * @param args not used
200         */
201        public static void main(String[] args) {
202                MidiExample instance = new MidiExample();
203                instance.run();
204        }
205}
